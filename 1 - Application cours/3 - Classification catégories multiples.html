# On d√©termine comment s√©parer les donn√©es d'entrainement et de validation (indiqu√© dans le fichier)

    train = df.index[~df['is_valid']].tolist()

# Dans le cas d'un probl√®me multicat√©gorie il nous faut adapter la fonction de calcul de performance. 
# Le but n'est plus de trouv√© la bonne cat√©gorie, il peut y en avoir plusieurs. Nous allons donc fixer un seuil √† partir 
# duquel on consid√©re qu'une cat√©gorie est "trouv√©e"

def accuracy_multi(inp, targ, thresh=0.5, sigmoid=True):
    if sigmoid: inp = inp.sigmoid() # On ramene la pr√©diction entre 0 et 1
    return ((inp>thresh) == targ.bool()).float().mean()

# Si la pr√©diction est sup√©rieur au seuil on consid√©re que la cat√©gories est bien √† trouver.
